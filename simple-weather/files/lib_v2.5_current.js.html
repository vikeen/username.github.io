                            "use strict";
                        
                        var config = require("../../config")(),
                            api = require("../api");
                        
                        /**
                         * @class current
                         */
                        module.exports = {
                            byCityName: byCityName,
                            byCityId: byCityId,
                            byCityIds: byCityIds,
                            byGeoLocation: byGeoLocation,
                            byZipcode: byZipcode
                        };
                        
                        /**
                         * My method description.  Like other pieces of your comment blocks,
                         * this can span multiple lines.
                         *
                         * @method byCityName
                         * @since 1.2.1
                         * @param {String} city The name of the city
                         * @param {String} country The ISO abbreviation of the country. Providing this improves specificity
                         * @return {Object} Returns city weather
                         *
                         * @example
                         * current.byCityName('Olathe');
                         * @example
                         * current.byCityName('Olathe', 'us');
                         */
                        function byCityName(city, country) {
                            var q = city;
                        
                            if (country) {
                                q += ',' + country
                            }
                        
                            return api.request({
                                method: "GET",
                                url: config.api["v2.5"].url + '/weather',
                                query: {
                                    q: q
                                }
                            });
                        }
                        
                        /**
